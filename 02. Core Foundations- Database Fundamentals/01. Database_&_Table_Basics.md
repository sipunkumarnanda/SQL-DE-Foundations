

CREATE TABLE `users` (
  user_id int AUTO_INCREMENT,
  name VARCHAR(30),
  email VARCHAR(30) NOT NULL UNIQUE,
  is_active BOOLEAN DEFAULT TRUE,
  account_balance DECIMAL,
  signup_date_time DATETIME,
  last_seen DATETIME,
  age INT NOT NULL,
  
  PRIMARY KEY (user_id),
  INDEX(email),
  CONSTRAINT age_check CHECK (((`age` >= 15) and (`age` <= 30)))
)
